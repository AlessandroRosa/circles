config set drawentity pixel
config set construction limitset
plugin set embedding general.purpose
plugin var g1_1=-1
plugin var g1_2=-2-2i
plugin var g1_3=2
plugin var g1_4=3+4i
plugin var g2_1=1
plugin var g2_2=-2i
plugin var g2_3=0

plugin var g3_1=3+2i

plugin var g3_3=-2
plugin var g3_4=-1-2i

plugin var g4_3=2+2i

auto refresh off
auto init off
zoom zplane (-0.75,0) side:2 silent
